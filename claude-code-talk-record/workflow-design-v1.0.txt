  ---
  关于工作流框架的整体架构，我提出两种方案：

  方案 A：单体引擎架构
  - 工作流配置和执行在同一个服务中完成
  - 执行引擎内置在 Spring Boot 应用中
  - 通过线程池管理并发执行
  - 优点：实现简单，部署方便
  - 缺点：执行引擎与业务耦合，难以扩展

  方案 B：分层解耦架构
  - 配置层：负责工作流定义、验证、存储
  - 调度层：负责执行调度、状态管理、WebSocket 推送
  - 执行层：独立的节点执行器，支持扩展
  - 优点：职责清晰，易于扩展和维护
  - 缺点：实现复杂度稍高

  我的推荐：方案 B（分层解耦架构）

  理由：考虑到未来可能需要支持更多节点类型、执行器扩展、以及分布式执行场景，分层
  设计更灵活。

  3.2.2 执行引擎

  /**
   * 执行引擎 - 负责工作流的编排和执行
   */
  @Service
  public class ExecutionEngine {

      private final NodeExecutorRegistry executorRegistry;
      private final StateManager stateManager;
      private final WebSocketManager wsManager;

      /**
       * 执行工作流
       */
      public void execute(WorkflowDefinition workflow, ExecutionContext context)
   {
          try {
              // 1. 构建执行图
              ExecutionGraph graph = buildExecutionGraph(workflow);

              // 2. 初始化开始节点
              List<WorkflowNode> startNodes = graph.getStartNodes();

              // 3. 执行节点
              executeNodes(graph, startNodes, context);

          } catch (Exception e) {
              handleExecutionError(context, e);
          }
      }

      /**
       * 执行节点集合
       */
      private void executeNodes(ExecutionGraph graph, 
                                List<WorkflowNode> nodes, 
                                ExecutionContext context) {
          while (!nodes.isEmpty()) {
              // 获取可执行节点（所有前置节点已完成）
              List<WorkflowNode> executableNodes = filterExecutableNodes(nodes,
  context);

              if (executableNodes.isEmpty() && !nodes.isEmpty()) {
                  // 死锁检测
                  throw new ExecutionException("检测到执行死锁");
              }

              // 并行执行可执行节点
              List<CompletableFuture<Void>> futures = executableNodes.stream()
                  .map(node -> CompletableFuture.runAsync(() ->
  executeNode(node, context)))
                  .collect(Collectors.toList());

              CompletableFuture.allOf(futures.toArray(new
  CompletableFuture[0])).join();

              // 移除已完成节点，添加后续节点
              nodes.removeAll(executableNodes);
              for (WorkflowNode node : executableNodes) {
                  nodes.addAll(graph.getNextNodes(node));
              }
          }
      }

      /**
       * 执行单个节点
       */
      private void executeNode(WorkflowNode node, ExecutionContext context) {
          NodeExecutor executor = executorRegistry.getExecutor(node.getType());

          // 更新状态：运行中
          stateManager.updateNodeStatus(context, node.getUuid(),
  NodeStatus.RUNNING);
          wsManager.pushNodeStatus(context.getExecutionUuid(), node.getUuid(),
  NodeStatus.RUNNING);

          try {
              // 解析输入参数
              Map<String, Object> inputs = resolveInputs(node, context);

              // 执行节点
              NodeExecutionResult result = executor.execute(node, inputs,
  context);

              // 存储输出
              context.setNodeOutputs(node.getUuid(), result.getOutputs());

              // 更新状态：成功
              stateManager.updateNodeStatus(context, node.getUuid(),
  NodeStatus.SUCCESS);
              wsManager.pushNodeStatus(context.getExecutionUuid(),
  node.getUuid(), NodeStatus.SUCCESS);

          } catch (Exception e) {
              // 错误处理
              handleNodeError(node, context, e);
          }
      }
  }

  context.getBodyNodes(node.getUuid());

          }

          // 用户定义的输出参数
          for (OutputParamConfig outParam : node.getLoopOutputParams()) {
              outputs.put(outParam.getName(), collectedOutputs);
          }

          return NodeExecutionResult.success(outputs);
      }

  异步处理节点执行器
  @Component
  public class AsyncNodeExecutor implements NodeExecutor {

      private final AsyncTaskManager asyncTaskManager;

      @Override
      public String getNodeType() {
          return "async";
      }

      @Override
      public NodeExecutionResult execute(WorkflowNode node,
                                         Map<String, Object> inputs,
                                         ExecutionContext context) {
          AsyncConfig config = node.getAsyncConfig();

          // 解析要处理的数组
          List<Object> items = resolveArraySource(config.getArraySource(),
  inputs);

          // 创建异步任务
          List<AsyncTaskInfo> tasks = new ArrayList<>();
          for (int i = 0; i < items.size(); i++) {
              AsyncTaskInfo task = asyncTaskManager.createTask(
                  context.getExecutionId(),
                  node.getUuid(),
                  items.get(i),
                  i,
                  config.getSubWorkflowId()
              );
              tasks.add(task);
          }

          // 注册异步任务到上下文
          context.registerAsyncTasks(node.getUuid(), tasks);

          // 触发异步执行（不等待）
          asyncTaskManager.triggerAsync(tasks, config.getConcurrency());

          // 立即返回，不阻塞
          return NodeExecutionResult.success(Map.of(
              "task_count", items.size(),

  @Component
  public class AIAgentClientExecutor implements SkillExecutor {

      private final ClientAgentProxy clientAgentProxy;

      @Override
      public ExecutionType getExecutionType() {
          return ExecutionType.AI_AGENT;
      }

      @Override
      public ExecutionLocation getExecutionLocation() {
          return ExecutionLocation.CLIENT;
      }

      @Override
      public SkillExecutionResult execute(SkillEntity skill,
                                          Map<String, Object> inputs,
                                          ExecutionContext context) {
          // 1. 查找可用的 Client 执行机
          ClientMachine client = clientAgentProxy.findAvailableClient(
              context.getTriggeredBy()
          );

          if (client == null) {
              throw new ExecutionException("无可用的 Client 执行机");
          }

          // 2. 构建任务请求
          AIAgentTaskRequest request = AIAgentTaskRequest.builder()
              .taskId(UUID.randomUUID().toString())
              .skillId(skill.getId())
              .skillName(skill.getName())
              .suitePath(skill.getSuitePath())
              .inputs(inputs)
              .executionConfig(buildAIExecutionConfig(skill))
              .build();

          // 3. 推送任务到 Client 端
          clientAgentProxy.pushTask(client, request);

          // 4. 等待执行结果（通过 WebSocket）
          SkillExecutionResult result = clientAgentProxy.waitForResult(
              request.getTaskId(),
              skill.getTimeout()
          );

          return result;
      }
  }

          return executor;
      }

          return type.name() + "_" + location.name();

  @Component
  public class SkillNodeExecutor implements NodeExecutor {

      private final SkillService skillService;
      private final SkillExecutorRegistry executorRegistry;

      @Override
      public String getNodeType() {
          return "skill";
      }

      @Override
      public NodeExecutionResult execute(WorkflowNode node,
                                         Map<String, Object> inputs,
                                         ExecutionContext context) {
          // 1. 获取 Skill 定义
          SkillEntity skill = skillService.getById(node.getSkillId());

          // 2. 参数校验与转换
          Map<String, Object> resolvedInputs = resolveAndValidateInputs(skill,
  inputs);

          // 3. 获取执行配置
          ExecutionType execType =
  ExecutionType.valueOf(skill.getExecutionType());
          ExecutionLocation execLocation = getExecutionLocation(node, skill);

          // 4. 获取对应执行器
          SkillExecutor executor = executorRegistry.getExecutor(execType,
  execLocation);

          // 5. 执行
          SkillExecutionResult result = executor.execute(skill, resolvedInputs,
  context);

          // 6. 处理结果
          if (!result.isSuccess()) {
              return handleExecutionFailure(node, result, context);
          }

          // 7. 映射输出参数
          Map<String, Object> outputs = mapOutputs(skill, result.getOutputs());

          return NodeExecutionResult.success(outputs);
      }

      /**
       * 获取执行位置
       * 优先级：节点配置 > Skill 配置 > 默认值(SERVICE)
       */
      private ExecutionLocation getExecutionLocation(WorkflowNode node, 
  SkillEntity skill) {
          // 节点级别配置优先
          if (node.getExecutionLocation() != null) {
              return ExecutionLocation.valueOf(node.getExecutionLocation());
          }
          // Skill 级别配置
          if (skill.getDefaultExecutionLocation() != null) {
              return
  ExecutionLocation.valueOf(skill.getDefaultExecutionLocation());
          }
          // 默认在 Service 端执行
          return ExecutionLocation.SERVICE;
      }
  }

  @Component
  public class AutomatedClientExecutor implements SkillExecutor {

      private final KafkaTaskProducer kafkaProducer;
      private final KafkaResultConsumer resultConsumer;
      private final ClientRegistry clientRegistry;

      @Override
      public ExecutionType getExecutionType() {
          return ExecutionType.AUTOMATED;
      }

      @Override
      public ExecutionLocation getExecutionLocation() {
          return ExecutionLocation.CLIENT;
      }

      @Override
      public SkillExecutionResult execute(SkillEntity skill,
                                          Map<String, Object> inputs,
                                          ExecutionContext context) {
          // 1. 选择可用的 Client 执行机
          ClientInfo client = selectAvailableClient(skill);

          if (client == null) {
              throw new ExecutionException("无可用的 Client 执行机");
          }

          // 2. 生成任务 ID
          String taskId = UUID.randomUUID().toString();

          // 3. 构建任务命令消息
          TaskCommandMessage command = TaskCommandMessage.builder()
              .messageId(UUID.randomUUID().toString())
              .messageType("TASK_COMMAND")
              .timestamp(System.currentTimeMillis())
              .payload(TaskCommandPayload.builder()
                  .taskId(taskId)
                  .executionId(context.getExecutionId())
                  .workflowId(context.getWorkflowId())
                  .nodeUuid(context.getCurrentNodeUuid())
                  .skillId(skill.getId())
                  .skillName(skill.getName())
                  .executionType(skill.getExecutionType())
                  .suitePath(skill.getSuitePath())
                  .inputs(inputs)
                  .timeout(skill.getTimeout())
                  .retryCount(skill.getRetryCount())
                  .priority("NORMAL")
                  .callbackTopic("task-client-results")
                  .build())
              .build();

          // 4. 注册结果等待器
          CompletableFuture<SkillExecutionResult> future =
              resultConsumer.registerPendingTask(taskId, skill.getTimeout());

          // 5. 发送任务到 Kafka（指定分区到目标 Client）
          kafkaProducer.sendTaskCommand(command, client.getClientId());

          // 6. 等待结果
          try {
              return future.get(skill.getTimeout(), TimeUnit.MILLISECONDS);
          } catch (TimeoutException e) {
              resultConsumer.removePendingTask(taskId);
              throw new ExecutionException("任务执行超时: " + taskId);
          } catch (InterruptedException | ExecutionException e) {
              resultConsumer.removePendingTask(taskId);
              throw new ExecutionException("任务执行异常: " + e.getMessage(),
  e);
          }
      }

      /**
       * 选择可用 Client
       */
      private ClientInfo selectAvailableClient(SkillEntity skill) {
          List<ClientInfo> availableClients =
  clientRegistry.getAvailableClients()
              .stream()
              .filter(c -> c.getSupportedExecutionTypes()
                  .contains(skill.getExecutionType()))
              .filter(c -> c.getRunningTasks() < c.getMaxConcurrency())
              .sorted(Comparator.comparingInt(ClientInfo::getRunningTasks))
              .collect(Collectors.toList());

          return availableClients.isEmpty() ? null : availableClients.get(0);
      }
  }

      private final KafkaTemplate<String, String> kafkaTemplate;

      private static final String TASK_COMMAND_TOPIC = "task-client-commands";

      /**
       * 发送任务命令到指定 Client
       */
      public void sendTaskCommand(TaskCommandMessage command, String clientId) {
          String partitionKey = clientId; // 使用 clientId 作为分区键

          String messageJson = toJson(command);

          CompletableFuture<SendResult<String, String>> future =
              kafkaTemplate.send(TASK_COMMAND_TOPIC, partitionKey, messageJson);

          future.whenComplete((result, ex) -> {
              if (ex != null) {
                  log.error("发送任务命令失败: taskId={}, clientId={}",
                      command.getPayload().getTaskId(), clientId, ex);
              } else {
                  log.info("任务命令已发送: taskId={}, partition={}, offset={}",
                      command.getPayload().getTaskId(),
                      result.getRecordMetadata().partition(),
                      result.getRecordMetadata().offset());
              }
          });
      }

  @Component
  public class KafkaResultConsumer {

      private final ConcurrentHashMap<String,
  CompletableFuture<SkillExecutionResult>>
          pendingTasks = new ConcurrentHashMap<>();

      private final ObjectMapper objectMapper;

      @KafkaListener(
          topics = "task-client-results",
          groupId = "workflow-engine-result-consumer",
          concurrency = "3"
      )
      public void consumeTaskResult(ConsumerRecord<String, String> record) {
          log.info("收到执行结果: partition={}, offset={}",
              record.partition(), record.offset());

          try {
              TaskResultMessage result = objectMapper.readValue(
                  record.value(),
                  TaskResultMessage.class
              );

              String taskId = result.getPayload().getTaskId();
              CompletableFuture<SkillExecutionResult> future =
  pendingTasks.get(taskId);

              if (future != null) {
                  SkillExecutionResult executionResult =
  SkillExecutionResult.builder()

  .success("SUCCESS".equals(result.getPayload().getStatus()))
                      .outputs(result.getPayload().getOutputs())
                      .logs(result.getPayload().getLogs())
                      .durationMs(result.getPayload().getDurationMs())
                      .errorMessage(result.getPayload().getErrorMessage())
                      .build();

                  future.complete(executionResult);
                  pendingTasks.remove(taskId);
              }

          } catch (Exception e) {
              log.error("处理执行结果失败", e);
          }
      }

      /**
       * 注册待处理任务
       */
      public CompletableFuture<SkillExecutionResult> registerPendingTask(
              String taskId, long timeoutMs) {
          CompletableFuture<SkillExecutionResult> future = new
  CompletableFuture<>();
          pendingTasks.put(taskId, future);

          // 超时自动移除
          ScheduledExecutorService scheduler =
  Executors.newSingleThreadScheduledExecutor();
          scheduler.schedule(() -> {
              CompletableFuture<SkillExecutionResult> removed =
  pendingTasks.remove(taskId);
              if (removed != null && !removed.isDone()) {
                  removed.completeExceptionally(new
  TimeoutException("任务等待超时"));
              }
          }, timeoutMs, TimeUnit.MILLISECONDS);

          return future;
      }

      public void removePendingTask(String taskId) {
          pendingTasks.remove(taskId);
      }
  }

  4.3 WebSocket 服务端实现

  @Component
  public class WebSocketHandler extends TextWebSocketHandler {

      private final ConcurrentHashMap<String, WebSocketSession> sessions = new
  ConcurrentHashMap<>();
      private final ConcurrentHashMap<String, Set<String>>
  executionSubscriptions = new ConcurrentHashMap<>();

      @Override
      public void afterConnectionEstablished(WebSocketSession session) {
          String sessionId = session.getId();
          sessions.put(sessionId, session);
          log.info("WebSocket 连接建立: sessionId={}", sessionId);
      }

      @Override
      public void afterConnectionClosed(WebSocketSession session, CloseStatus 
  status) {
          String sessionId = session.getId();
          sessions.remove(sessionId);
          executionSubscriptions.remove(sessionId);
          log.info("WebSocket 连接关闭: sessionId={}", sessionId);
      }

      /**
       * 处理客户端消息
       */
      @Override
      protected void handleTextMessage(WebSocketSession session, TextMessage 
  message) {
          WebSocketMessage wsMessage = parseMessage(message.getPayload());

          switch (wsMessage.getType()) {
              case "SUBSCRIBE_EXECUTION":
                  handleSubscribe(session, wsMessage);
                  break;
              case "UNSUBSCRIBE_EXECUTION":
                  handleUnsubscribe(session, wsMessage);
                  break;
              default:
                  log.warn("未知消息类型: {}", wsMessage.getType());
          }
      }

      /**
       * 订阅执行状态
       */
      private void handleSubscribe(WebSocketSession session, WebSocketMessage 
  message) {
          String executionUuid =
  message.getPayload().get("executionUuid").toString();

          executionSubscriptions
              .computeIfAbsent(session.getId(), k ->
  ConcurrentHashMap.newKeySet())
              .add(executionUuid);

          sendMessage(session, WebSocketMessage.builder()
              .type("SUBSCRIBE_SUCCESS")
              .payload(Map.of("executionUuid", executionUuid))
              .build());
      }

      /**
       * 推送消息给订阅了指定执行的所有客户端
       */
      public void broadcastToExecution(String executionUuid, WebSocketMessage 
  message) {
          String messageJson = toJson(message);

          executionSubscriptions.forEach((sessionId, subscriptions) -> {
              if (subscriptions.contains(executionUuid)) {
                  WebSocketSession session = sessions.get(sessionId);
                  if (session != null && session.isOpen()) {
                      sendMessage(session, messageJson);
                  }
              }
          });
      }

      private void sendMessage(WebSocketSession session, String message) {
          try {
              session.sendMessage(new TextMessage(message));
          } catch (IOException e) {
              log.error("发送 WebSocket 消息失败: sessionId={}",
  session.getId(), e);
          }
      }
  }

  5.3 前置节点计算算法

  @Component
  public class PredecessorCalculator {

      /**
       * 计算指定节点的所有前置节点
       * @param workflow 工作流定义
       * @param nodeUuid 目标节点UUID
       * @return 前置节点UUID集合（按执行顺序排序）
       */
      public List<String> calculatePredecessors(WorkflowDefinition workflow, 
  String nodeUuid) {
          Set<String> visited = new HashSet<>();
          List<String> result = new ArrayList<>();

          // 构建邻接表（反向图）
          Map<String, List<String>> reverseGraph = buildReverseGraph(workflow);

          // DFS 遍历获取所有前置节点
          dfs(nodeUuid, reverseGraph, visited, result);

          // 移除自身
          result.remove(nodeUuid);

          return result;
      }

      /**
       * 构建反向邻接表
       */
      private Map<String, List<String>> buildReverseGraph(WorkflowDefinition
  workflow) {
          Map<String, List<String>> graph = new HashMap<>();

          // 初始化所有节点
          for (WorkflowNode node : workflow.getNodes()) {
              graph.put(node.getNodeUuid(), new ArrayList<>());
          }

          // 根据连线构建反向图
          for (WorkflowConnection conn : workflow.getConnections()) {
              String target = conn.getTargetNodeUuid();
              String source = conn.getSourceNodeUuid();

              graph.computeIfAbsent(target, k -> new ArrayList<>()).add(source);
          }

          return graph;
      }

      /**
       * DFS 遍历
       */
      private void dfs(String nodeUuid, 
                       Map<String, List<String>> graph,
                       Set<String> visited,
                       List<String> result) {
          if (visited.contains(nodeUuid)) {
              return;
          }
          visited.add(nodeUuid);

          List<String> predecessors = graph.getOrDefault(nodeUuid,
  Collections.emptyList());
          for (String pred : predecessors) {
              dfs(pred, graph, visited, result);
          }

          result.add(nodeUuid);
      }

      /**
       * 获取前置节点的所有输出变量（供前端选择器使用）
       */
      public List<VariableOption> getAvailableVariables(WorkflowDefinition 
  workflow, 
                                                         String targetNodeUuid) 
  {
          List<String> predecessors = calculatePredecessors(workflow,
  targetNodeUuid);
          List<VariableOption> options = new ArrayList<>();

          for (String nodeUuid : predecessors) {
              WorkflowNode node = workflow.getNode(nodeUuid);
              if (node == null) continue;

              // 获取节点输出参数
              List<ParameterConfig> outputs = getOutputParams(node);

              for (ParameterConfig param : outputs) {
                  options.add(VariableOption.builder()
                      .nodeUuid(nodeUuid)
                      .nodeName(node.getNodeName())
                      .paramName(param.getName())
                      .paramType(param.getType())
                      .fullReference("${" + node.getNodeName() + "." +
  param.getName() + "}")
                      .build());
              }
          }

          return options;
      }
  }

  5.4.3 兼容性检查服务

  @Service
  public class SkillCompatibilityChecker {

      private final WorkflowNodeMapper nodeMapper;

      /**
       * 检查 Skill 变更的兼容性
       */
      @Transactional
      public CompatibilityCheckResult checkCompatibility(
              String skillId, 
              SkillEntity oldSkill,
              SkillEntity newSkill) {

          // 查询所有引用该 Skill 的节点
          List<WorkflowNodeEntity> affectedNodes =
  nodeMapper.findBySkillId(skillId);

          CompatibilityCheckResult result = new CompatibilityCheckResult();
          result.setTotalAffectedNodes(affectedNodes.size());

          for (WorkflowNodeEntity node : affectedNodes) {
              NodeCompatibilityStatus status = checkNodeCompatibility(
                  node,
                  oldSkill,
                  newSkill
              );

              // 更新节点状态
              node.setCompatibilityStatus(status.name());
              nodeMapper.updateById(node);

              result.addResult(node.getId(), status);
          }

          return result;
      }

      /**
       * 检查单个节点的兼容性
       */
      private NodeCompatibilityStatus checkNodeCompatibility(
              WorkflowNodeEntity node,
              SkillEntity oldSkill,
              SkillEntity newSkill) {

          List<SkillParameterEntity> oldInputs = getParameters(oldSkill,
  "INPUT");
          List<SkillParameterEntity> newInputs = getParameters(newSkill,
  "INPUT");

          // 解析节点当前配置
          Map<String, Object> currentConfig =
  parseInputConfig(node.getInputConfig());

          // 检查删除的必填参数
          for (SkillParameterEntity oldParam : oldInputs) {
              if (oldParam.getRequired()) {
                  boolean stillExists = newInputs.stream()
                      .anyMatch(p ->
  p.getParamName().equals(oldParam.getParamName()));
                  if (!stillExists) {
                      return NodeCompatibilityStatus.INCOMPATIBLE;
                  }
              }
          }

          // 检查新增的必填参数
          for (SkillParameterEntity newParam : newInputs) {
              if (newParam.getRequired()) {
                  boolean isConfigured =
  currentConfig.containsKey(newParam.getParamName());
                  if (!isConfigured) {
                      return NodeCompatibilityStatus.NEEDS_UPDATE;
                  }
              }
          }

          // 检查类型变更
          for (SkillParameterEntity newParam : newInputs) {
              SkillParameterEntity oldParam = oldInputs.stream()
                  .filter(p -> p.getParamName().equals(newParam.getParamName()))
                  .findFirst()
                  .orElse(null);

              if (oldParam != null &&
  !oldParam.getParamType().equals(newParam.getParamType())) {
                  return NodeCompatibilityStatus.INCOMPATIBLE;
              }
          }

          return NodeCompatibilityStatus.COMPATIBLE;
      }
  }

  5.5 工作流保存前验证

  @Service
  public class WorkflowValidator {

      /**
       * 验证工作流定义
       */
      public ValidationResult validate(WorkflowDefinition workflow) {
          ValidationResult result = new ValidationResult();

          // 1. 结构验证
          validateStructure(workflow, result);

          // 2. 节点配置验证
          validateNodes(workflow, result);

          // 3. 连线验证
          validateConnections(workflow, result);

          // 4. 参数引用验证
          validateParameterReferences(workflow, result);

          // 5. Skill 节点兼容性验证
          validateSkillCompatibility(workflow, result);

          return result;
      }

      /**
       * 验证结构完整性
       */
      private void validateStructure(WorkflowDefinition workflow, 
  ValidationResult result) {
          // 必须有且仅有一个开始节点
          long startCount = workflow.getNodes().stream()
              .filter(n -> "start".equals(n.getNodeType()))
              .count();
          if (startCount == 0) {
              result.addError("工作流必须包含一个开始节点");
          } else if (startCount > 1) {
              result.addError("工作流只能包含一个开始节点");
          }

          // 必须有且仅有一个结束节点
          long endCount = workflow.getNodes().stream()
              .filter(n -> "end".equals(n.getNodeType()))
              .count();
          if (endCount == 0) {
              result.addError("工作流必须包含一个结束节点");
          } else if (endCount > 1) {
              result.addError("工作流只能包含一个结束节点");
          }

          // 不能有孤立节点
          Set<String> connectedNodes = new HashSet<>();
          for (WorkflowConnection conn : workflow.getConnections()) {
              connectedNodes.add(conn.getSourceNodeUuid());
              connectedNodes.add(conn.getTargetNodeUuid());
          }

          for (WorkflowNode node : workflow.getNodes()) {
              if (!connectedNodes.contains(node.getNodeUuid())) {
                  result.addWarning("节点 [" + node.getNodeName() + "] 
  未连接到任何其他节点");
              }
          }
      }

      /**
       * 验证节点配置
       */
      private void validateNodes(WorkflowDefinition workflow, ValidationResult 
  result) {
          for (WorkflowNode node : workflow.getNodes()) {
              // Skill 节点必须关联 Skill
              if ("skill".equals(node.getNodeType())) {
                  if (node.getSkillId() == null || node.getSkillId().isEmpty())
  {
                      result.addError("节点 [" + node.getNodeName() + "] 未关联 
  Skill");
                  }
              }

              // 循环节点必须有循环体
              if ("loop".equals(node.getNodeType())) {
                  boolean hasBody = workflow.getNodes().stream()
                      .anyMatch(n ->
  node.getNodeUuid().equals(n.getParentNodeUuid()));
                  if (!hasBody) {
                      result.addWarning("循环节点 [" + node.getNodeName() + "] 
  没有配置循环体");
                  }
              }

              // 验证必填参数是否配置
              validateRequiredParams(node, result);
          }
      }

      /**
       * 验证参数引用有效性
       */
      private void validateParameterReferences(WorkflowDefinition workflow, 
  ValidationResult result) {
          PredecessorCalculator calculator = new PredecessorCalculator();

          for (WorkflowNode node : workflow.getNodes()) {
              if ("start".equals(node.getNodeType())) continue;

              List<String> predecessors = calculator.calculatePredecessors(
                  workflow,
                  node.getNodeUuid()
              );

              // 检查输入配置中的所有引用
              List<ParameterConfig> inputParams =
  node.getInputConfig().getParameters();
              for (ParameterConfig param : inputParams) {
                  if ("reference".equals(param.getValueSource().getType())) {
                      String ref = param.getValueSource().getValue();
                      String referencedNode = extractNodeName(ref);

                      // 检查引用的节点是否是前置节点
                      boolean isValidRef = predecessors.stream()
                          .anyMatch(uuid -> {
                              WorkflowNode pred = workflow.getNode(uuid);
                              return pred != null &&
  pred.getNodeName().equals(referencedNode);
                          });

                      if (!isValidRef) {
                          result.addError("节点 [" + node.getNodeName() + "] 
  的参数 [" +
                              param.getName() + "] 引用了非前置节点的输出: " +
  ref);
                      }
                  }
              }
          }
      }
  }

  @Component
  public class LoopNodeExecutor implements NodeExecutor {

      @Override
      public String getNodeType() {
          return "loop";
      }

      @Override
      public NodeExecutionResult execute(WorkflowNode node,
                                         Map<String, Object> inputs,
                                         ExecutionContext context) {
          LoopConfig config = node.getLoopConfig();
          List<Object> items = resolveLoopItems(config, inputs, context);

          // 安全检查：最大迭代次数
          int maxIterations = config.getMaxIterations() != null ?
              config.getMaxIterations() : 1000;
          if (items.size() > maxIterations) {
              throw new ExecutionException(
                  "循环次数超过最大限制: " + items.size() + " > " +
  maxIterations
              );
          }

          // 收集循环输出
          Map<String, List<Object>> collectedOutputs = new HashMap<>();
          for (OutputParamConfig outParam : config.getOutputParams()) {
              collectedOutputs.put(outParam.getName(), new ArrayList<>());
          }

          // 顺序执行循环体
          for (int i = 0; i < items.size(); i++) {
              // 检查工作流是否被中止
              if (context.isAborted()) {
                  return NodeExecutionResult.aborted();
              }

              // 设置循环变量
              context.setLoopVariable(node.getNodeUuid(), "current_item",
  items.get(i));
              context.setLoopVariable(node.getNodeUuid(), "current_index", i);

              // 推送进度
              int progress = (int) ((i + 1) * 100.0 / items.size());
              context.publishProgress(node.getNodeUuid(), progress,
                  "执行第 " + (i + 1) + "/" + items.size() + " 次迭代");

              // 执行循环体
              List<WorkflowNode> bodyNodes =
  context.getBodyNodes(node.getNodeUuid());
              executeLoopBody(bodyNodes, context);

              // 收集输出
              for (OutputParamConfig outParam : config.getOutputParams()) {
                  Object value = context.resolveReference(outParam.getSource());
                  collectedOutputs.get(outParam.getName()).add(value);
              }
          }

          // 构建输出
          Map<String, Object> outputs = new HashMap<>();
          for (Map.Entry<String, List<Object>> entry :
  collectedOutputs.entrySet()) {
              outputs.put(entry.getKey(), entry.getValue());
          }

          return NodeExecutionResult.success(outputs);
      }

      /**
       * 解析循环项
       */
      private List<Object> resolveLoopItems(LoopConfig config,
                                            Map<String, Object> inputs,
                                            ExecutionContext context) {
          switch (config.getType()) {
              case "COUNT":
                  int times = config.getTimes();
                  return IntStream.range(0, times)
                      .mapToObj(i -> i)
                      .collect(Collectors.toList());

              case "ARRAY":
                  Object array =
  context.resolveReference(config.getArraySource().getValue());
                  if (array instanceof List) {
                      return (List<Object>) array;
                  }
                  throw new ExecutionException("数组遍历源不是有效的数组: " +
  array);

              case "CONDITION":
                  // 条件循环：先加入一个元素，后续在执行时检查条件
                  return Collections.singletonList(true);

              default:
                  throw new ExecutionException("未知的循环类型: " +
  config.getType());
          }
      }
  }

  6.2.2 批处理节点执行器

  @Component
  public class BatchNodeExecutor implements NodeExecutor {

      private final ExecutorService batchExecutor =
  Executors.newCachedThreadPool();

      @Override
      public String getNodeType() {
          return "batch";
      }

      @Override
      public NodeExecutionResult execute(WorkflowNode node,
                                         Map<String, Object> inputs,
                                         ExecutionContext context) {
          BatchConfig config = node.getBatchConfig();

          // 解析数组
          List<Object> items =
  context.resolveReference(config.getArraySource().getValue());
          if (!(items instanceof List)) {
              throw new ExecutionException("批处理源不是有效的数组");
          }

          List<Object> itemList = (List<Object>) items;
          int concurrency = config.getConcurrency();

          // 创建信号量控制并发
          Semaphore semaphore = new Semaphore(concurrency);

          // 并发执行
          List<CompletableFuture<BatchTaskResult>> futures = new ArrayList<>();
          List<BatchTaskResult> results = Collections.synchronizedList(new
  ArrayList<>());
          AtomicInteger successCount = new AtomicInteger(0);
          AtomicInteger failedCount = new AtomicInteger(0);

          for (int i = 0; i < itemList.size(); i++) {
              final int index = i;
              final Object item = itemList.get(i);

              CompletableFuture<BatchTaskResult> future =
  CompletableFuture.supplyAsync(() -> {
                  try {
                      semaphore.acquire();

                      // 推送任务开始
                      context.publishBatchProgress(node.getNodeUuid(),
                          index, itemList.size(), "RUNNING");

                      // 创建子执行上下文
                      ExecutionContext subContext = context.createSubContext();
                      subContext.setLoopVariable(node.getNodeUuid(),
  "current_item", item);
                      subContext.setLoopVariable(node.getNodeUuid(),
  "current_index", index);

                      // 执行批处理体
                      List<WorkflowNode> bodyNodes =
  context.getBodyNodes(node.getNodeUuid());
                      executeBatchBody(bodyNodes, subContext);

                      // 收集结果
                      Map<String, Object> taskOutput = collectTaskOutput(config,
   subContext);
                      successCount.incrementAndGet();

                      BatchTaskResult result = BatchTaskResult.success(index,
  taskOutput);
                      results.add(result);

                      return result;

                  } catch (InterruptedException e) {
                      Thread.currentThread().interrupt();
                      failedCount.incrementAndGet();
                      return BatchTaskResult.failed(index, "任务被中断");

                  } catch (Exception e) {
                      failedCount.incrementAndGet();

                      // 根据失败策略处理
                      if ("STOP_ALL".equals(config.getFailureStrategy())) {
                          context.abort();
                      }

                      return BatchTaskResult.failed(index, e.getMessage());

                  } finally {
                      semaphore.release();
                  }
              }, batchExecutor);

              futures.add(future);
          }

          // 等待所有任务完成
          CompletableFuture.allOf(futures.toArray(new
  CompletableFuture[0])).join();

          // 构建输出
          List<Object> mergedOutputs = results.stream()
              .filter(BatchTaskResult::isSuccess)
              .map(BatchTaskResult::getOutput)
              .collect(Collectors.toList());

          Map<String, Object> outputs = new HashMap<>();
          outputs.put("batch_results", mergedOutputs);
          outputs.put("total_count", itemList.size());
          outputs.put("success_count", successCount.get());
          outputs.put("failed_count", failedCount.get());

          // 根据失败策略决定最终状态
          if (failedCount.get() > 0 &&
  "STOP_ALL".equals(config.getFailureStrategy())) {
              return NodeExecutionResult.failure("批处理任务部分失败", outputs);
          }

          return NodeExecutionResult.success(outputs);
      }
  }

  @Component
  public class AsyncNodeExecutor implements NodeExecutor {

      private final KafkaTaskProducer kafkaProducer;
      private final AsyncTaskRegistry asyncTaskRegistry;

      @Override
      public String getNodeType() {
          return "async";
      }

      @Override
      public NodeExecutionResult execute(WorkflowNode node,
                                         Map<String, Object> inputs,
                                         ExecutionContext context) {
          AsyncConfig config = node.getAsyncConfig();

          // 解析数组
          List<Object> items;
          if ("ARRAY".equals(config.getType())) {
              items =
  context.resolveReference(config.getArraySource().getValue());
          } else {
              // SINGLE 模式：单次执行
              items = Collections.singletonList(inputs);
          }

          // 创建异步任务列表
          List<AsyncTaskInfo> tasks = new ArrayList<>();

          for (int i = 0; i < items.size(); i++) {
              String taskId = UUID.randomUUID().toString();

              AsyncTaskInfo task = AsyncTaskInfo.builder()
                  .taskId(taskId)
                  .asyncNodeUuid(node.getNodeUuid())
                  .itemIndex(i)
                  .itemData(items.get(i))
                  .status("PENDING")
                  .createdAt(LocalDateTime.now())
                  .build();

              tasks.add(task);

              // 发送任务到 Kafka（不等待结果）
              TaskCommandMessage command = buildTaskCommand(task, config,
  context);
              kafkaProducer.sendTaskCommand(command, null); // 不指定 
  clientId，由任意可用执行机消费
          }

          // 注册异步任务到上下文
          context.registerAsyncTasks(node.getNodeUuid(), tasks);

          // 立即返回，不阻塞
          return NodeExecutionResult.success(Map.of(
              "task_count", items.size(),
              "async_triggered", true,
              "task_ids",
  tasks.stream().map(AsyncTaskInfo::getTaskId).collect(Collectors.toList())
          ));
      }
  }

  6.3.2 结果收集节点执行器

  @Component
  public class CollectNodeExecutor implements NodeExecutor {

      private final AsyncTaskRegistry asyncTaskRegistry;
      private final KafkaResultConsumer kafkaResultConsumer;

      @Override
      public String getNodeType() {
          return "collect";
      }

      @Override
      public NodeExecutionResult execute(WorkflowNode node,
                                         Map<String, Object> inputs,
                                         ExecutionContext context) {
          CollectConfig config = node.getCollectConfig();

          // 获取关联的异步任务
          List<AsyncTaskInfo> tasks =
  context.getAsyncTasks(config.getAsyncNodeId());

          if (tasks == null || tasks.isEmpty()) {
              throw new ExecutionException("未找到关联的异步任务: " +
  config.getAsyncNodeId());
          }

          // 根据等待策略处理
          List<AsyncTaskResult> completedResults = new ArrayList<>();
          int waitCount = getWaitCount(config, tasks.size());
          int successCount = 0;
          int failedCount = 0;

          long startTime = System.currentTimeMillis();
          long timeout = config.getTimeout();

          while (completedResults.size() < waitCount) {
              // 检查超时
              if (System.currentTimeMillis() - startTime > timeout) {
                  break;
              }

              // 检查工作流是否被中止
              if (context.isAborted()) {
                  return NodeExecutionResult.aborted();
              }

              // 查询任务状态
              for (AsyncTaskInfo task : tasks) {
                  if (task.getStatus().equals("COMPLETED") &&
  !task.isCollected()) {
                      AsyncTaskResult result =
  asyncTaskRegistry.getTaskResult(task.getTaskId());
                      if (result != null) {
                          completedResults.add(result);
                          task.setCollected(true);

                          if (result.isSuccess()) {
                              successCount++;
                          } else {
                              failedCount++;
                          }
                      }
                  }
              }

              // 等待一小段时间再轮询
              try {
                  Thread.sleep(500);
              } catch (InterruptedException e) {
                  Thread.currentThread().interrupt();
                  break;
              }
          }

          // 处理输出
          List<Object> outputs = processOutputs(completedResults, config);

          Map<String, Object> result = new HashMap<>();
          result.put("async_results", outputs);
          result.put("total_count", tasks.size());
          result.put("collected_count", completedResults.size());
          result.put("success_count", successCount);
          result.put("failed_count", failedCount);
          result.put("timeout", System.currentTimeMillis() - startTime >
  timeout);

          // 根据失败处理策略决定最终状态
          return determineFinalStatus(config, successCount, failedCount,
  result);
      }

      private int getWaitCount(CollectConfig config, int totalTasks) {
          switch (config.getWaitStrategy()) {
              case "ALL":
                  return totalTasks;
              case "FIRST":
                  return 1;
              case "N":
                  return Math.min(config.getWaitCount(), totalTasks);
              default:
                  return totalTasks;
          }
      }

      private List<Object> processOutputs(List<AsyncTaskResult> results, 
  CollectConfig config) {
          switch (config.getOutputMode()) {
              case "MERGE":
                  return results.stream()
                      .filter(AsyncTaskResult::isSuccess)
                      .map(AsyncTaskResult::getOutput)
                      .collect(Collectors.toList());

              case "SEPARATE":
                  return results.stream()
                      .map(r -> Map.of(
                          "success", r.isSuccess(),
                          "output", r.getOutput(),
                          "error", r.getError()
                      ))
                      .collect(Collectors.toList());

              case "FIRST_SUCCESS":
                  return results.stream()
                      .filter(AsyncTaskResult::isSuccess)
                      .findFirst()
                      .map(r -> Collections.singletonList(r.getOutput()))
                      .orElse(Collections.emptyList());

              default:
                  return Collections.emptyList();
          }
      }
  }

  6.4.2 条件节点执行器

  @Component
  public class ConditionNodeExecutor implements NodeExecutor {

      @Override
      public String getNodeType() {
          return "condition_simple";
      }

      @Override
      public NodeExecutionResult execute(WorkflowNode node,
                                         Map<String, Object> inputs,
                                         ExecutionContext context) {
          ConditionConfig config = node.getConditionConfig();
          Expression expression = config.getExpression();

          // 解析操作数
          Object leftValue = resolveOperand(expression.getLeftOperand(),
  context);
          Object rightValue = resolveOperand(expression.getRightOperand(),
  context);

          // 计算条件结果
          boolean result = evaluateCondition(
              leftValue,
              expression.getOperator(),
              rightValue
          );

          // 返回分支标签
          String branchLabel = result ? "true" : "false";

          return NodeExecutionResult.branchResult(branchLabel, Map.of(
              "condition_result", result,
              "branch", branchLabel
          ));
      }

      /**
       * 解析操作数
       */
      private Object resolveOperand(Object operand, ExecutionContext context) {
          if (operand == null) {
              return null;
          }

          if (operand instanceof String) {
              String str = (String) operand;
              if (str.startsWith("${") && str.endsWith("}")) {
                  // 变量引用
                  return context.resolveReference(str);
              }
              return str;
          }

          return operand;
      }

      /**
       * 计算条件
       */
      private boolean evaluateCondition(Object left, String operator, Object 
  right) {
          switch (operator) {
              // 字符串操作
              case "equals":
                  return Objects.equals(left, right);
              case "notEquals":
                  return !Objects.equals(left, right);
              case "contains":
                  return left != null &&
  left.toString().contains(right.toString());
              case "startsWith":
                  return left != null &&
  left.toString().startsWith(right.toString());
              case "endsWith":
                  return left != null &&
  left.toString().endsWith(right.toString());
              case "isEmpty":
                  return left == null || left.toString().isEmpty();
              case "isNotEmpty":
                  return left != null && !left.toString().isEmpty();

              // 数值操作
              case "greaterThan":
                  return compareNumbers(left, right) > 0;
              case "lessThan":
                  return compareNumbers(left, right) < 0;
              case "greaterThanOrEqual":
                  return compareNumbers(left, right) >= 0;
              case "lessThanOrEqual":
                  return compareNumbers(left, right) <= 0;
              case "between":
                  Object[] range = (Object[]) right;
                  return compareNumbers(left, range[0]) >= 0 &&
                         compareNumbers(left, range[1]) <= 0;

              // 布尔操作
              case "isTrue":
                  return Boolean.TRUE.equals(left);
              case "isFalse":
                  return Boolean.FALSE.equals(left);

              // 数组操作
              case "sizeEquals":
                  return left instanceof List && ((List<?>) left).size() ==
  ((Number) right).intValue();
              case "sizeGreaterThan":
                  return left instanceof List && ((List<?>) left).size() >
  ((Number) right).intValue();

              default:
                  throw new ExecutionException("未知的操作符: " + operator);
          }
      }

      private int compareNumbers(Object a, Object b) {
          double aVal = ((Number) a).doubleValue();
          double bVal = ((Number) b).doubleValue();
          return Double.compare(aVal, bVal);
      }
  }
