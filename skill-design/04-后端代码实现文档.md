```java
package com.example.demo.skill.entity;

import com.baomidou.mybatisplus.annotation.*;
import lombok.Data;
import java.io.Serializable;
import java.time.LocalDateTime;

@Data
@TableName("skill")
public class SkillEntity implements Serializable {

    @TableId(value = "id", type = IdType.ASSIGN_UUID)
    private String id;

    @TableField("name")
    private String name;                    // Skill名称, 全局唯一

    @TableField("description")
    private String description;             // 描述

    @TableField("suite_path")
    private String suitePath;               // 执行套件文件路径

    @TableField("suite_filename")
    private String suiteFilename;           // 执行套件原始文件名

    @TableField("execution_type")
    private String executionType;           // 执行类型: AUTOMATED/AI

    @TableField("category")
    private String category;                // 分类: SYSTEM/USER

    @TableField("access_type")
    private String accessType;              // 访问控制: PUBLIC/PRIVATE/WHITELIST/PROJECT

    @TableField("is_container")
    private Boolean isContainer;            // 是否容器

    @TableField("allow_add_input_params")
    private Boolean allowAddInputParams;    // 是否支持增加入参

    @TableField("allow_add_output_params")
    private Boolean allowAddOutputParams;   // 是否支持增加出参

    @TableField("status")
    private String status;                  // 状态: PUBLISHED/DRAFT

    @TableField("created_by")
    private String createdBy;               // 创建人

    @TableField("updated_by")
    private String updatedBy;               // 更新人

    @TableField(value = "created_at", fill = FieldFill.INSERT)
    private LocalDateTime createdAt;        // 创建时间

    @TableField(value = "updated_at", fill = FieldFill.INSERT_UPDATE)
    private LocalDateTime updatedAt;        // 更新时间

    @TableLogic
    @TableField("deleted")
    private Boolean deleted;                // 逻辑删除标记

    @TableField("deleted_at")
    private LocalDateTime deletedAt;        // 删除时间
}
```

```java
@Override
@Transactional
public SkillResponse createSkill(SkillCreateRequest request, MultipartFile file) {
    log.info("创建Skill: {}", request.getName());

    // 1. 检查名称唯一性
    Long count = skillMapper.countByName(request.getName());
    if (count > 0) {
        throw BusinessException.conflict("Skill名称已存在: " + request.getName());
    }

    // 2. 处理文件上传
    String suitePath = null;
    String suiteFilename = null;
    if (file != null && !file.isEmpty()) {
        suitePath = saveSuiteFile(file);
        suiteFilename = file.getOriginalFilename();
    }

    // 3. 创建Skill主表记录
    SkillEntity skill = new SkillEntity();
    skill.setName(request.getName());
    skill.setDescription(request.getDescription());
    skill.setSuitePath(suitePath);
    skill.setSuiteFilename(suiteFilename);
    skill.setExecutionType(request.getExecutionType());
    skill.setCategory(request.getCategory());
    skill.setAccessType(request.getAccessType());
    skill.setIsContainer(request.getIsContainer());
    skill.setAllowAddInputParams(request.getAllowAddInputParams() != null ? request.getAllowAddInputParams() : false);
    skill.setAllowAddOutputParams(request.getAllowAddOutputParams() != null ? request.getAllowAddOutputParams() : false);
    skill.setStatus(SkillStatus.DRAFT.name());
    skill.setCreatedBy(createdBy);
    skill.setUpdatedBy(createdBy);
    skill.setDeleted(false);

    skillMapper.insert(skill);
    String skillId = skill.getId();

    // 4. 保存入参
    if (request.getInputParameters() != null) {
        int order = 1;
        for (InputParameterDTO paramDTO : request.getInputParameters()) {
            ParameterTypeValidator.validateType(paramDTO.getParamType(), paramDTO.getParamName());
            SkillParameterEntity param = new SkillParameterEntity();
            param.setSkillId(skillId);
            param.setParamDirection(SkillParamDirection.INPUT.name());
            param.setParamOrder(order++);
            param.setParamType(paramDTO.getParamType());
            param.setParamName(paramDTO.getParamName());
            param.setDefaultValue(paramDTO.getDefaultValue());
            param.setDescription(paramDTO.getDescription());
            param.setRequired(paramDTO.getRequired());
            parameterMapper.insert(param);
        }
    }

    // 5. 保存出参
    // ... 类似入参处理

    // 6. 保存访问控制
    // ...

    return getSkillById(skillId);
}
```

```java
@Override
@Transactional
public SkillResponse updateSkill(String id, SkillUpdateRequest request, MultipartFile file) {
    SkillEntity skill = skillMapper.selectById(id);
    if (skill == null || skill.getDeleted()) {
        throw BusinessException.notFound("Skill", id);
    }

    // 1. 检查名称唯一性(如果修改了名称)
    if (request.getName() != null && !request.getName().equals(skill.getName())) {
        Long count = skillMapper.countByName(request.getName());
        if (count > 0) {
            throw BusinessException.conflict("Skill名称已存在");
        }
        skill.setName(request.getName());
    }

    // 2. 更新基本字段
    if (request.getDescription() != null) skill.setDescription(request.getDescription());
    if (file != null && !file.isEmpty()) {
        String suitePath = saveSuiteFile(file);
        skill.setSuitePath(suitePath);
        skill.setSuiteFilename(file.getOriginalFilename());
    }
    // ... 其他字段

    skillMapper.updateById(skill);

    // 3. 更新入参(先删后插)
    if (request.getInputParameters() != null) {
        parameterMapper.deleteBySkillIdAndDirection(id, SkillParamDirection.INPUT);
        // 插入新参数...
    }

    // 4. 更新出参
    // ...

    return getSkillById(id);
}
```

```java
public SkillResponse copySkill(String id) {
    SkillEntity original = skillMapper.selectById(id);
    if (original == null || original.getDeleted()) {
        throw BusinessException.notFound("Skill", id);
    }

    // 1. 创建新Skill
    SkillEntity newSkill = new SkillEntity();

    newSkill.setIsContainer(original.getIsContainer());
    newSkill.setStatus(SkillStatus.DRAFT.name());  // 复制后始终为草稿
    newSkill.setCreatedBy(original.getCreatedBy());
    newSkill.setUpdatedBy(original.getCreatedBy());
    newSkill.setDeleted(false);

    skillMapper.insert(newSkill);
    String newSkillId = newSkill.getId();

    // 2. 复制所有参数
    List<SkillParameterEntity> params = parameterMapper.selectBySkillId(id);
    for (SkillParameterEntity param : params) {
        SkillParameterEntity newParam = new SkillParameterEntity();
        // 复制属性...
        newParam.setSkillId(newSkillId);
        parameterMapper.insert(newParam);
    }

    // 3. 复制访问控制
    List<SkillAccessControlEntity> accessControls = accessControlMapper.selectBySkillId(id);
    for (SkillAccessControlEntity ac : accessControls) {
        SkillAccessControlEntity newAc = new SkillAccessControlEntity();
        // 复制属性...
        newAc.setSkillId(newSkillId);
        accessControlMapper.insert(newAc);
    }
```

#### 文件保存

```java
@Value("${skill.suite.upload-path:./uploads/suites}")
private String uploadPath;

private String saveSuiteFile(MultipartFile file) {
    try {
        // 1. 创建上传目录
        Path uploadDir = Paths.get(uploadPath);
        if (!Files.exists(uploadDir)) {
            Files.createDirectories(uploadDir);
        }

        // 2. 生成唯一文件名
        String originalFilename = file.getOriginalFilename();
        String extension = "";
        if (originalFilename != null && originalFilename.contains(".")) {
            extension = originalFilename.substring(originalFilename.lastIndexOf("."));
        }
        String newFilename = UUID.randomUUID().toString() + extension;

        // 3. 保存文件
        Path filePath = uploadDir.resolve(newFilename);
        Files.copy(file.getInputStream(), filePath);

        return filePath.toString();
    } catch (IOException e) {
        throw BusinessException.invalidParam("执行套件文件保存失败: " + e.getMessage());
    }
}
```

## 5. 参数类型验证器

### 5.1 ParameterTypeValidator.java

```java
package com.example.demo.skill.util;

import com.example.demo.common.exception.BusinessException;
import java.util.List;
import java.util.regex.Pattern;

public class ParameterTypeValidator {

    // 基本类型
    private static final List<String> BASIC_TYPES = List.of(
        "String", "Boolean", "Integer", "Object", "Times"
    );

    // 文件类型
    private static final List<String> FILE_TYPES = List.of(
        "Zip", "Doc", "Docx", "Excel", "Pdf", "Txt"
    );

    // 数组元素类型
    private static final List<String> ARRAY_ELEMENT_TYPES = List.of(
        "String", "Boolean", "Integer", "Time", "Object"
    );

    // 正则表达式
    private static final Pattern ARRAY_PATTERN = Pattern.compile("^Array<([A-Za-z]+)>$");
    private static final Pattern FILE_PATTERN = Pattern.compile("^File<(.+)>$");
    private static final Pattern ARRAY_FILE_PATTERN = Pattern.compile("^Array<File<(.+)>>$");

    /**
     * 验证参数类型是否有效
     */
    public static boolean isValidType(String type) {
        if (type == null || type.isEmpty()) return false;

        // 1. 检查基本类型
        if (BASIC_TYPES.contains(type)) return true;

        // 2. 检查 Array<File<FileType>>
        var arrayFileMatcher = ARRAY_FILE_PATTERN.matcher(type);
        if (arrayFileMatcher.matches()) {
            return FILE_TYPES.contains(arrayFileMatcher.group(1));
        }

        // 3. 检查 Array<ElementType>
        var arrayMatcher = ARRAY_PATTERN.matcher(type);
        if (arrayMatcher.matches()) {
            String elementType = arrayMatcher.group(1);
            return ARRAY_ELEMENT_TYPES.contains(elementType) || "File".equals(elementType);
        }

        // 4. 检查 File<FileType>
        var fileMatcher = FILE_PATTERN.matcher(type);
        if (fileMatcher.matches()) {
            return FILE_TYPES.contains(fileMatcher.group(1));
        }

        // 5. 检查单独的 Array 或 File
        return "Array".equals(type) || "File".equals(type);
    }

    /**
     * 验证并抛出异常
     */
    public static void validateType(String type, String paramName) {
        if (!isValidType(type)) {
            throw BusinessException.invalidParam(
                String.format("参数 '%s' 的类型 '%s' 无效", paramName, type)
            );
        }
    }
}
```

## 6. 定时任务

### 6.1 SkillAgingCleanupScheduler.java

```java
package com.example.demo.skill.scheduler;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;

@Slf4j
@Component
@RequiredArgsConstructor
public class SkillAgingCleanupScheduler {

    private final SkillMapper skillMapper;
    private final SkillParameterMapper parameterMapper;
    private final SkillAccessControlMapper accessControlMapper;

    private static final int AGING_DAYS = 365;  // 老化阈值

    /**
     * 每天凌晨0点5分执行
     */
    @Scheduled(cron = "0 5 0 * * ?")
    public void cleanupAgedSkillData() {
        log.info("开始执行Skill数据老化清理任务...");

        try {
            // 1. 查询已删除超过365天的Skill
            List<SkillEntity> agedSkills = skillMapper.selectDeletedOlderThan(AGING_DAYS);
            log.info("发现已删除超过{}天的Skill数量: {}", AGING_DAYS, agedSkills.size());

            if (agedSkills.isEmpty()) {
                log.info("没有需要清理的Skill数据");
                return;
            }

            // 2. 逐个物理删除
            int deletedCount = 0;
            for (SkillEntity skill : agedSkills) {
                try {
                    // 删除关联参数
                    parameterMapper.deleteBySkillId(skill.getId());
                    // 删除访问控制
                    accessControlMapper.deleteBySkillId(skill.getId());
                    // 物理删除Skill
                    skillMapper.physicalDeleteById(skill.getId());
                    deletedCount++;
                    log.info("已物理删除Skill: ID={}, Name={}", skill.getId(), skill.getName());
                } catch (Exception e) {
                    log.error("物理删除Skill失败: ID={}", skill.getId(), e);
                }
            }

            log.info("Skill数据老化清理完成，共删除 {} 条记录", deletedCount);
        } catch (Exception e) {
            log.error("Skill数据老化清理任务失败", e);
        }
    }
}
```

## 9. 事务管理

所有写操作都使用 `@Transactional` 注解:

- `createSkill`: 创建操作, 失败时回滚所有插入
- `updateSkill`: 更新操作, 失败时回滚
- `deleteSkill`: 删除操作, 失败时回滚
- `copySkill`: 复制操作, 失败时回滚
- `publishSkill`/`unpublishSkill`: 状态变更
